# AI Clinical Board v2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Upgrade AI Clinical Board with grounded prompting (per-agent domain sheets), mode-specific agents (cancer: treatment-focused), clinical note input, and a hybrid Wiki+SQLite knowledge base for decision history.

**Architecture:** Three phases — (1) domain sheets + cancer agents with config-driven temperature, (2) clinical note CLI input, (3) knowledge base with SQLite storage + Wiki generation. All phases modify `runner.py` as the central orchestrator.

**Tech Stack:** Python 3.10+, SQLite, MedGemma 27B via Ollama, Jinja2, pytest

**Spec:** `docs/superpowers/specs/2026-04-14-ai-board-v2-design.md`

# scripts/clinical_board/domain_sheets.py
"""Build per-agent, per-mode domain sheets from local DB data."""

MAX_DOMAIN_CHARS = 16_000  # ~4K tokens

def build_domain_sheet(agent_domain: str, mode: str, variants: list, report_data: dict) -> str:
    """Build domain-specific data sheet for an agent."""
    if mode == "cancer":
        builders = _CANCER_BUILDERS
    else:
        builders = _RARE_DISEASE_BUILDERS

    builder = builders.get(agent_domain, _empty_sheet)
    sheet = builder(variants, report_data)

    if len(sheet) > MAX_DOMAIN_CHARS:
        sheet = sheet[:MAX_DOMAIN_CHARS] + "\n[...truncated for context limit]"
    return sheet

def _empty_sheet(variants, report_data):
    return ""

def _rd_variant_pathologist(variants, report_data):
    # Build ClinVar + in silico detail for each variant
    ...

def _rd_disease_geneticist(variants, report_data):
    # Build OMIM + GeneReviews + HPO + Orphanet detail
    ...

def _rd_pgx_specialist(variants, report_data):
    # Build CPIC guideline + Korean prevalence
    ...

def _rd_literature_analyst(variants, report_data):
    # Build CIViC evidence + GeneReviews + ClinGen
    ...

_RARE_DISEASE_BUILDERS = {
    "variant_pathology": _rd_variant_pathologist,
    "disease_genetics": _rd_disease_geneticist,
    "pharmacogenomics": _rd_pgx_specialist,
    "literature_evidence": _rd_literature_analyst,
}
```

Each builder iterates `variants` list and formats relevant fields into readable text sections. Use data already present in `report_data["variants"]` — no new DB queries needed since pipeline already collected this data.

def _build_clinical_note_section(data: dict) -> Optional[str]:

# scripts/db/build_kb_db.py
"""Initialize Knowledge Base SQLite schema."""
import sqlite3
from pathlib import Path

def build_kb_db(db_path: str = "data/knowledge_base/kb.sqlite3"):
    Path(db_path).parent.mkdir(parents=True, exist_ok=True)
    conn = sqlite3.connect(db_path)
    conn.execute("""CREATE TABLE IF NOT EXISTS board_decisions (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        sample_id TEXT NOT NULL,
        mode TEXT NOT NULL,
        date TEXT NOT NULL,
        gene TEXT NOT NULL,
        variant TEXT NOT NULL,
        hgvsc TEXT,
        hgvsp TEXT,
        classification TEXT,
        board_diagnosis TEXT,
        board_confidence TEXT,
        clinical_context_summary TEXT,
        agent_consensus TEXT,
        raw_opinion_json TEXT
    )""")
    conn.execute("CREATE INDEX IF NOT EXISTS idx_bd_variant ON board_decisions(gene, variant, mode)")
    conn.execute("CREATE INDEX IF NOT EXISTS idx_bd_gene ON board_decisions(gene, mode)")
    conn.execute("CREATE INDEX IF NOT EXISTS idx_bd_date ON board_decisions(date)")
    conn.execute("CREATE INDEX IF NOT EXISTS idx_bd_hgvsp ON board_decisions(gene, hgvsp, mode)")
    conn.execute("DROP VIEW IF EXISTS variant_stats")
    conn.execute("""CREATE VIEW variant_stats AS
        SELECT gene, variant, mode,
            COUNT(*) as total_cases,
            SUM(CASE WHEN board_confidence = 'high' THEN 1 ELSE 0 END) as high_confidence_count,
            GROUP_CONCAT(DISTINCT board_diagnosis) as diagnoses_seen,
            MAX(date) as last_seen
        FROM board_decisions GROUP BY gene, variant, mode""")
    conn.commit()
    conn.close()
```
